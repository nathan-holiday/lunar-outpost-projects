#timestamps for cmap labels, will use colormap for progress
num_ts = 7 # will add the last one at the end so actually will have +1
length = len(df.time)
indices = [int((i*length)/num_ts) for i in range(num_ts)]
timestamps = [round(df.time[i]) for i in indices]
timestamps.append(round(df.time.iloc[-1]))
labels = [str(time//60) + 'm ' + str(time%60)+'s' for time in timestamps]
ticks = np.linspace(0,1,num_ts+1)

ax = fig.add_subplot(111, projection = '3d')

z = df.z
colors = np.linspace(0,1,len(df.x))


ax.plot(x, y, z, color = 'black', alpha = 0.6, label = 'rover_path')
sc = ax.scatter(
    x, y, z,
    c=colors,
    cmap='gist_rainbow',
)
ax.set(
    xlabel = 'x (meters)',
    ylabel = 'y (meters)',
    zlabel = 'z (meters)',
    title = 'Rover Path 3d (height changes exaggerated 20x)'
)
cbar=fig.colorbar(sc, ax=ax, shrink=0.55, pad = 0.12)
cbar.set_ticks(ticks, labels=labels)
